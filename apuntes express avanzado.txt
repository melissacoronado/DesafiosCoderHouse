app.patch('/productos/:id/title', (req: Request, res: Response) => {  //o quitar title y pasarlo como json - probar si sirve put y patch
    try{    
        const id: number = +req.params.id //Viene de la url/1 y el + para parsear a numero
		const product =  opsProd.productos.find(x => x.id === id)
		
		if (product === null)
			res.status(404).json({error : 'Producto no encontrado'})
		
		const { title, price, thumbnail } = req.body  
		product.title = title
		product.price = price
		product.thumbnail = thumbnail		

        res.status(204).send("Producto Modificado!")        
    }catch(error){
        res.status(404).json({error : 'No se pudo agregar el Producto.'})
    }
})
